#+BEGIN_SRC c++
  constexpr 
  struct AddM : Lambda<AddM> {
  
    constexpr static mx = var<0>;
    constexpr static my = var<1>;
    constexpr static x  = var<2>;
    constexpr static y  = var<3>;

    constexpr static proc =
      fn( mx, my,
	  (letm( x = mx,
		 y = my ),
	   rtn(x + y)));
  } addM{};
